Quick Start

# Install
bun install

# Configure
echo 'ANTHROPIC_API_KEY=sk-ant-...' > .env

# Run
bun src/cli.ts # Interactive TUI
bun src/cli.ts query "hello" # Quick query

# With Feature Flags
SAWCODE_ENABLE_FILE_TOOLS=false bun src/cli.ts
DEBUG=query-handler bun src/cli.ts query "test"
